#include <iostream>
using namespace std;

int main()
{
	int tab[10][10], o, w;


	for (int o = 0; o < 10; o++)

		for (int w = 0; w < 10; w++)
		{
			if (o == w)
			{
				tab[o][w] = 1;
			}
			else
			{
				tab[o][w] = 0;
			}
			cout << tab[o][w] << " ";
		}
	cout << endl;
}
cout << endl;


return 0;
}
